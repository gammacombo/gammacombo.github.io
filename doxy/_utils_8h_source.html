155 template<typename T>
156 std::vector<T> Utils::Quantile(const std::vector<T>& inData, const std::vector<T>& probs)
157 {
158  if (inData.empty())
159  {
160  return std::vector<T>();
161  }
162 
163  if (1 == inData.size())
164  {
165  return std::vector<T>(1, inData[0]);
166  }
167 
168  std::vector<T> data = inData;
169  std::sort(data.begin(), data.end());
170  std::vector<T> quantiles;
171 
172  for (size_t i = 0; i < probs.size(); ++i)
173  {
174  T poi = Lerp<T>(-0.5, data.size() - 0.5, probs[i]);
175 
176  size_t left = std::max(int64_t(std::floor(poi)), int64_t(0));
177  size_t right = std::min(int64_t(std::ceil(poi)), int64_t(data.size() - 1));
178 
179  T datLeft = data.at(left);
180  T datRight = data.at(right);
181 
182  T quantile = Lerp<T>(datLeft, datRight, poi - left);
183 
184  quantiles.push_back(quantile);
185  }
186 
187  return quantiles;
188 }
