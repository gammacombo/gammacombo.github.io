141  string line;

145  if ( line.empty() ) continue; // blank line

149  startingValues.push_back(map<TString,double>());

151  else {

153  boost::split(els,line,boost::is_any_of(" "),boost::token_compress_on);
154  TString name = els[0];
155  double val = boost::lexical_cast<double>(els[1]);
156  startingValues[nSolutions-1].insert(make_pair(name,val));
157  }
158  }
159  }
160  m_parametersLoaded=true;
161  successfullyLoaded=true;
162  if ( m_arg->debug ) printPoint();
163  }
164  else {
165  cout << "ParameterCache::loadPoints() : ERROR : file not found: " << fileName << endl;
166  }
