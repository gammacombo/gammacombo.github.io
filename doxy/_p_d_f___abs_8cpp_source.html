253 void PDF_Abs::buildCov()
254 {
255  // add diagonals, symmetrize
256  buildCorMatrix(corStatMatrix);
257  buildCorMatrix(corSystMatrix);
258 
259  // make total cov matrix
260  TMatrixDSym *covStat = buildCovMatrix(corStatMatrix, StatErr);
261  TMatrixDSym *covSyst = buildCovMatrix(corSystMatrix, SystErr);
262  covMatrix = *covStat + *covSyst;
263 
264  // check if total cov matrix is invertible
265  if ( covMatrix.Determinant()==0 ) {
266  cout << "PDF_Abs::buildCov() : ERROR : Total covariance matrix is not invertable (det(COV)=0)." << endl;
267  cout << "PDF_Abs::buildCov() : ERROR : Check inputs! Ordering correct? Nobs correct?" << endl;
268  cout << "PDF_Abs::buildCov() : PDF: " << name << endl;
269  cout << "PDF_Abs::buildCov() : stat cov: " << endl;
270  covStat->Print("v");
271  cout << "PDF_Abs::buildCov() : syst cov: " << endl;
272  covSyst->Print("v");
273  cout << "PDF_Abs::buildCov() : full cov: " << endl;
274  covMatrix.Print("v");
275  //exit(1);
276  throw TString("need help");
277  }
278 
279  // make total cor matrix
280  for ( int i=0; i<covMatrix.GetNcols(); i++ )
281  for ( int j=0; j<covMatrix.GetNcols(); j++ )
282  {
283  corMatrix[i][j] = covMatrix[i][j]/sqrt(covMatrix[i][i])/sqrt(covMatrix[j][j]);
284  }
285 
286  // check if total cor matrix is positive definite
287  if ( ! isPosDef(&corMatrix) ) {
288  cout << "PDF_Abs::buildCov() : ERROR : Total correlation matrix is not positive definite." << endl;
289  cout << "PDF_Abs::buildCov() : ERROR : Check inputs! Ordering correct?" << endl;
290  cout << "PDF_Abs::buildCov() : Sometimes this happens when for very large correlations" << endl;
291  cout << "PDF_Abs::buildCov() : the given precision is not enough (e.g. rho=0.98 rather than 0.978)." << endl;
292  cout << "PDF_Abs::buildCov() : PDF: " << name << endl;
293  cout << "PDF_Abs::buildCov() : stat cor: " << endl;
294  corStatMatrix.Print("v");
295  cout << "PDF_Abs::buildCov() : syst cor: " << endl;
296  corSystMatrix.Print("v");
297  //exit(1);
298  throw TString("need help");
299  }
300 
301  delete covStat;
302  delete covSyst;
303 
304  // this is needed for the pull computation and the PDF_Abs::print() function:
305  storeErrorsInObs();

534  TString pThName = p->GetName();

540  }
541 
542  // check if the 'observables' and 'theory' lists are correctly ordered
543  for ( int i=0; i<nObs; i++ ){
544  RooAbsArg* pTh = theory->at(i);
545  TString base = pTh->GetName();
546  base.ReplaceAll("_th","");

553  cout << " Check ordering of the 'theory' and 'observables' lists!" << endl;
