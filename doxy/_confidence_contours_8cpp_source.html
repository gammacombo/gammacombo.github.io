65 TH2F* ConfidenceContours::addBoundaryBins(TH2F* hist)
66 {
67  float boundary = hist->GetMinimum();
68  TH2F* hBoundaries = new TH2F(getUniqueRootName(),getUniqueRootName(),
69  hist->GetNbinsX()+2,
70  hist->GetXaxis()->GetXmin() - hist->GetXaxis()->GetBinWidth(1),
71  hist->GetXaxis()->GetXmax() + hist->GetXaxis()->GetBinWidth(1),
72  hist->GetNbinsY()+2,
73  hist->GetYaxis()->GetXmin() - hist->GetYaxis()->GetBinWidth(1),
74  hist->GetYaxis()->GetXmax() + hist->GetYaxis()->GetBinWidth(1));
75  for ( int ix=1; ix<=hBoundaries->GetXaxis()->GetNbins(); ix++ ){
76  for ( int iy=1; iy<=hBoundaries->GetYaxis()->GetNbins(); iy++ ){
77  if ( ix==1 || ix==hBoundaries->GetXaxis()->GetNbins()
78  || iy==1 || iy==hBoundaries->GetYaxis()->GetNbins() )
79  hBoundaries->SetBinContent(ix,iy,boundary);
80  else{
81  hBoundaries->SetBinContent(ix,iy,hist->GetBinContent(ix-1,iy-1));
82  }
83  }
84  }
85  return hBoundaries;
86 }

144 void ConfidenceContours::computeContours(TH2F* hist, histogramType type, int id)
145 {
146  if ( m_arg->debug ) {
147  cout << "ConfidenceContours::computeContours() : making contours of histogram ";
148  cout << hist->GetName();
149  cout << ", type " << (type==kChi2?"chi2":"p-value") << endl;
150  }
151  // clean up contours from a previous call
152  m_contours.clear();
153 
154  // transform chi2 from valley to hill
155  float offset = 30.;
156  if ( type==kChi2 ) hist = transformChi2valleyToHill(hist,offset);
157 
158  // add boundaries
159  TH2F* histb = addBoundaryBins(hist);
160 
161  // make contours
162  const int nMaxContours = 5;
163  histb->SetContour(nMaxContours);
164  if ( type==kChi2 ) {
165  // chi2 units
166  if ( m_arg->plot2dcl[id]>0 ){
167  histb->SetContourLevel(4, offset- 2.30);
168  histb->SetContourLevel(3, offset- 6.18);
169  histb->SetContourLevel(2, offset-11.83);
170  histb->SetContourLevel(1, offset-19.34);
171  histb->SetContourLevel(0, offset-28.76);
172  }
173  else{
174  histb->SetContourLevel(4, offset-1.);
175  histb->SetContourLevel(3, offset-4.);
176  histb->SetContourLevel(2, offset-9.);
177  histb->SetContourLevel(1, offset-16.);
178  histb->SetContourLevel(0, offset-25.);
179  }
180  }
181  else {
182  // p-value units
183  if ( m_arg->plot2dcl[id]>0 ){
184  histb->SetContourLevel(4, 0.3173);
185  histb->SetContourLevel(3, 4.55e-2);
186  histb->SetContourLevel(2, 2.7e-3);
187  histb->SetContourLevel(1, 6.3e-5);
188  histb->SetContourLevel(0, 5.7e-7);
189  }
190  else{
191  histb->SetContourLevel(4, 1.-0.39);
192  histb->SetContourLevel(3, 1.-0.87);
193  histb->SetContourLevel(2, 1.-0.989);
194  histb->SetContourLevel(1, 1.-0.9997);
195  histb->SetContourLevel(0, 1.-0.999997);
196  }
197  }
198 
199  // create and access the contours
200  if ( m_arg->interactive ) gROOT->SetBatch(true); // don't display the temporary canvas
201  TCanvas *ctmp = newNoWarnTCanvas(getUniqueRootName(), "ctmp");
202  histb->Draw("contlist");
203  gPad->Update(); // needed to be able to access the contours as TGraphs
204  TObjArray *contours = (TObjArray*)gROOT->GetListOfSpecials()->FindObject("contours");
205  delete ctmp;
206  delete histb;
207  if ( m_arg->interactive ) gROOT->SetBatch(false); // it's important to only unset batch mode if we're interactive! Else some canvases get screwed up badly resulting in corrupted PDF files.
208 
209  // access contours. They get filled in reverse order,
210  // and depend on how many are actually present. If all 5
211  // are filled, index 0 is 5sigma. If only 2 are filled, index 0
212  // is 2 sigma.
213  int nEmptyContours = 0;
214  for ( int ic=4; ic>=0; ic-- ){
215  if (((TList*)contours->At(ic))->IsEmpty()) nEmptyContours++;
216  }
217  for ( int ic=4; ic>=0; ic-- ){
218  if ( !(((TList*)contours->At(ic))->IsEmpty()) ){
219  Contour* cont = new Contour(m_arg, (TList*)contours->At(ic));
220  cont->setSigma(5-nEmptyContours-ic);
221  m_contours.push_back(cont);
222  }
223  }
224 
225  // add the entire plotted area, if one requested contour
226  // is empty, i.e. it contains the entire plot range
227  if ( nEmptyContours>0 ){
228  addFilledPlotArea(hist);
229  }
230 
231  // magnetic boundaries
232  if ( m_arg->plotmagnetic ) {
233  for ( int ic=4; ic>=0; ic-- ){
234  if ( ic>=m_contours.size() ) continue;
235  m_contours[ic]->magneticBoundaries(hist);
236  }
237  }
238 
239  // clean up
240  if ( type==kChi2 ) delete hist; // a copy was made earlier in this case
241 }
