93 template<class BASE>
94 RooBinned2DBicubicBase<BASE>::RooBinned2DBicubicBase(
95  const char* name, const char* title, const TH2& h,
96  RooAbsReal& xvar, RooAbsReal& yvar) :
97  BASE(name, title),
98  x("x", "x", this, xvar), y("y", "y", this, yvar),
99  nBinsX(1 + h.GetNbinsX()), nBinsY(1 + h.GetNbinsY()),
100  binSizeX(h.GetXaxis()->GetBinWidth(1)),
101  binSizeY(h.GetYaxis()->GetBinWidth(1)),
102  xmin(h.GetXaxis()->GetBinCenter(1) - binSizeX),
103  xmax(h.GetXaxis()->GetBinCenter(nBinsX - 1) + binSizeX),
104  ymin(h.GetYaxis()->GetBinCenter(1) - binSizeY),
105  ymax(h.GetYaxis()->GetBinCenter(nBinsY - 1) + binSizeY),
106  coeffs(CoeffRecLen * nBinsX * nBinsY)
107 {
108  const TAxis *xaxis = h.GetXaxis(), *yaxis = h.GetYaxis();
109  // verify that all bins have same size
110  for (int i = 1; i < nBinsX; ++i) {
111  if (std::abs(xaxis->GetBinWidth(i) / binSizeX - 1.) > 1e-9)
112  throw BinSizeException();
113  }
114  for (int i = 1; i < nBinsY; ++i) {
115  if (std::abs(yaxis->GetBinWidth(i) / binSizeY - 1.) > 1e-9)
116  throw BinSizeException();
117  }
118  // ok, go through histogram to precalculate the interpolation coefficients
119  // in rectangles between bin centres
120  //
121  // for that purpose, we map each of those rectangles to the unit square
122  for (int j = -1; j < nBinsY - 1; ++j) {
123  for (int i = -1; i < nBinsX - 1; ++i) {
124  const double rhs[NCoeff] = {
125  // function values in bin centres
126  histcont(h, i, j),
127  histcont(h, i + 1, j),
128  histcont(h, i, j + 1),
129  histcont(h, i + 1, j + 1),
130  // df/dx in bin centres (finite difference approximation)
131  dhistdx(h, i, j),
132  dhistdx(h, i + 1, j),
133  dhistdx(h, i, j + 1),
134  dhistdx(h, i + 1, j + 1),
135  // df/dy in bin centres (finite difference approximation)
136  dhistdy(h, i, j),
137  dhistdy(h, i + 1, j),
138  dhistdy(h, i, j + 1),
139  dhistdy(h, i + 1, j + 1),
140  // d^2f/dxdy in bin centres (finite difference approximation)
141  d2histdxdy(h, i, j),
142  d2histdxdy(h, i + 1, j),
143  d2histdxdy(h, i, j + 1),
144  d2histdxdy(h, i + 1, j + 1)
145  };
146  // work out solution - strange array placement is due to the fact
147  // that terms with x/y to high powers can be small, so they should
148  // be added up first during evaluation to avoid cancellation
149  // issues; at the same time you want to access them in order to
150  // not confuse the CPU cache, so they're stored back to front
151  //
152  // a_00 ... a_30
153  coeff(1 + i, 1 + j, 15) = rhs[0];
154  coeff(1 + i, 1 + j, 14) = rhs[4];
155  coeff(1 + i, 1 + j, 13) =
156  3. * (-rhs[0] + rhs[1]) - 2. * rhs[4] - rhs[5];
157  coeff(1 + i, 1 + j, 12) =
158  2. * (rhs[0] - rhs[1]) + rhs[4] + rhs[5];
159  // a_31 ... a_31
160  coeff(1 + i, 1 + j, 11) = rhs[8];
161  coeff(1 + i, 1 + j, 10) = rhs[12];
162  coeff(1 + i, 1 + j, 9) =
163  3. * (-rhs[8] + rhs[9]) - 2. * rhs[12] - rhs[13];
164  coeff(1 + i, 1 + j, 8) =
165  2. * (rhs[8] - rhs[9]) + rhs[12] + rhs[13];
166  // a_02 ... a_32
167  coeff(1 + i, 1 + j, 7) =
168  3. * (-rhs[0] + rhs[2]) - 2. * rhs[8] - rhs[10];
169  coeff(1 + i, 1 + j, 6) =
170  3. * (-rhs[4] + rhs[6]) - 2. * rhs[12] - rhs[14];
171  coeff(1 + i, 1 + j, 5) =
172  9. * (rhs[0] - rhs[1] - rhs[2] + rhs[3]) +
173  6. * (rhs[4] - rhs[6] + rhs[8] - rhs[9]) + 4. * rhs[12] +
174  3. * (rhs[5] - rhs[7] + rhs[10] - rhs[11]) +
175  2. * (rhs[13] + rhs[14]) + rhs[15];
176  coeff(1 + i, 1 + j, 4) =
177  6. * (-rhs[0] + rhs[1] + rhs[2] - rhs[3]) +
178  4. * (-rhs[8] + rhs[9]) +
179  3. * (-rhs[4] - rhs[5] + rhs[6] + rhs[7]) +
180  2. * (-rhs[10] + rhs[11] - rhs[12] - rhs[13]) -
181  rhs[14] - rhs[15];
182  // a_03 ... a_33
183  coeff(1 + i, 1 + j, 3) =
184  2. * (rhs[0] - rhs[2]) + rhs[8] + rhs[10];
185  coeff(1 + i, 1 + j, 2) =
186  2. * (rhs[4] - rhs[6]) + rhs[12] + rhs[14];
187  coeff(1 + i, 1 + j, 1) =
188  6. * (-rhs[0] + rhs[1] + rhs[2] - rhs[3]) +
189  4. * (-rhs[4] + rhs[6]) +
190  3. * (-rhs[8] + rhs[9] - rhs[10] + rhs[11]) +
191  2. * (- rhs[5] + rhs[7] - rhs[12] - rhs[14]) -
192  rhs[13] - rhs[15];
193  coeff(1 + i, 1 + j, 0) =
194  4. * (rhs[0] - rhs[1] - rhs[2] + rhs[3]) +
195  2. * (rhs[4] + rhs[5] - rhs[6] - rhs[7] +
196  rhs[8] - rhs[9] + rhs[10] - rhs[11]) +
197  rhs[12] + rhs[13] + rhs[14] + rhs[15];
198  // coeff(1 + i, 1 + j, 17) contains integral of function over the
199  // square in "unit square coordinates", i.e. neglecting bin widths
200  // this is done to help speed up calculations of 2D integrals
201  double sum = 0.;
202  for (int k = 0; k < NCoeff; ++k)
203  sum += coeff(1 + i, 1 + j, k) /
204  double((4 - (k % 4)) * (4 - (k / 4)));
205  coeff(1 + i, 1 + j, NCoeff) = sum;
206  }
207  }

254 template<class BASE>
255 double RooBinned2DBicubicBase<BASE>::eval(double x, double y) const
256 {
257  if (x != x || y != y) return 0.;
258  if (x <= xmin || x >= xmax || y <= ymin || y >= ymax) return 0.;
259  // find the bin in question
260  const int binx = int(double(nBinsX) * (x - xmin) / (xmax - xmin));
261  const int biny = int(double(nBinsY) * (y - ymin) / (ymax - ymin));
262  // get low edge of bin
263  const double xlo = double(nBinsX - binx) / double(nBinsX) * xmin +
264  double(binx) / double(nBinsX) * xmax;
265  const double ylo = double(nBinsY - biny) / double(nBinsY) * ymin +
266  double(biny) / double(nBinsY) * ymax;
267  // normalise to coordinates in unit sqare
268  const double hx = (x - xlo) / binSizeX;
269  const double hy = (y - ylo) / binSizeY;
270  // monomials
271  const double hxton[4] = { hx * hx * hx, hx * hx, hx, 1. };
272  const double hyton[4] = { hy * hy * hy, hy * hy, hy, 1. };
273  // sum up
274  double retVal = 0.;
275  for (int k = 0; k < NCoeff; ++k)
276  retVal += coeff(binx, biny, k) * hxton[k % 4] * hyton[k / 4];
277 
278  return retVal;
279 }

281 template<class BASE>
282 double RooBinned2DBicubicBase<BASE>::evalX(double x1, double x2, double y) const
283 {
284  if (x1 != x1 || x2 != x2 || y != y) return 0.;
285  // find the bin in question
286  const int biny = int(double(nBinsY) * (y - ymin) / (ymax - ymin));
287  // get low edge of bin
288  const double ylo = double(nBinsY - biny) / double(nBinsY) * ymin +
289  double(biny) / double(nBinsY) * ymax;
290  // normalise to coordinates in unit sqare
291  const double hy = (y - ylo) / binSizeY;
292  // monomials
293  const double hyton[4] = { hy * hy * hy, hy * hy, hy, 1. };
294  // integral
295  double sum = 0.;
296  for (int binx = 0; binx < nBinsX; ++binx) {
297  // get low/high edge of bin
298  const double xhi = double(nBinsX - binx - 1) / double(nBinsX) * xmin +
299  double(binx + 1) / double(nBinsX) * xmax;
300  if (xhi < x1) continue;
301  const double xlo = double(nBinsX - binx) / double(nBinsX) * xmin +
302  double(binx) / double(nBinsX) * xmax;
303  if (xlo > x2) break;
304  // work out integration range
305  const double a = ((xlo > x1) ? 0. : (x1 - xlo)) / binSizeX;
306  const double b = ((xhi < x2) ? binSizeX : (x2 - xlo)) / binSizeX;
307  // integrated monomials
308  const double hxton[4] = { 0.25 * (b * b * b * b - a * a * a * a),
309  (b * b * b - a * a * a) / 3., 0.5 * (b * b - a * a), b - a };
310  double lsum = 0.;
311  for (int k = 0; k < NCoeff; ++k)
312  lsum += coeff(binx, biny, k) * hxton[k % 4] * hyton[k / 4];
313  sum += lsum;
314  }
315  // move from unit square coordinates to user coordinates
316  return sum * binSizeX;
317 }

357 template<class BASE>
358 double RooBinned2DBicubicBase<BASE>::evalXY(
359  double x1, double x2, double y1, double y2) const
360 {
361  if (x1 != x1 || y1 != y1) return 0.;
362  if (x2 != x2 || y2 != y2) return 0.;
363  // integral
364  double sum = 0.;
365  for (int biny = 0; biny < nBinsY; ++biny) {
366  // get low/high edge of bin
367  const double yhi = double(nBinsY - biny - 1) / double(nBinsY) * ymin +
368  double(biny + 1) / double(nBinsY) * ymax;
369  if (yhi < y1) continue;
370  const double ylo = double(nBinsY - biny) / double(nBinsY) * ymin +
371  double(biny) / double(nBinsY) * ymax;
372  if (ylo > y2) break;
373  // work out integration range
374  const double ay = ((ylo > y1) ? 0. : (y1 - ylo)) / binSizeY;
375  const double by = ((yhi < y2) ? binSizeY : (y2 - ylo)) / binSizeY;
376  const bool yFullyContained = std::abs(by - ay - 1.0) < 1e-15;
377  // integrated monomials
378  const double hyton[4] = {
379  0.25 * (by * by * by * by - ay * ay * ay * ay),
380  (by * by * by - ay * ay * ay) / 3., 0.5 * (by * by - ay * ay),
381  by - ay };
382  for (int binx = 0; binx < nBinsX; ++binx) {
383  // get low/high edge of bin
384  const double xhi = double(nBinsX - binx - 1) / double(nBinsX) * xmin +
385  double(binx + 1) / double(nBinsX) * xmax;
386  if (xhi < x1) continue;
387  const double xlo = double(nBinsX - binx) / double(nBinsX) * xmin +
388  double(binx) / double(nBinsX) * xmax;
389  if (xlo > x2) break;
390  // work out integration range
391  const double ax = ((xlo > x1) ? 0. : (x1 - xlo)) / binSizeX;
392  const double bx = ((xhi < x2) ? binSizeX : (x2 - xlo)) / binSizeX;
393  const bool xFullyContained = std::abs(bx - ax - 1.0) < 1e-15;
394  if (xFullyContained && yFullyContained) {
395  // for fully contained bins, we have cached the integral
396  sum += coeff(binx, biny, NCoeff);
397  continue;
398  }
399  // integrated monomials
400  const double hxton[4] = {
401  0.25 * (bx * bx * bx * bx - ax * ax * ax * ax),
402  (bx * bx * bx - ax * ax * ax) / 3., 0.5 * (bx * bx - ax * ax),
403  bx - ax };
404  // integrate over bin
405 
406  double lsum = 0.;
407  for (int k = 0; k < NCoeff; ++k)
408  lsum += coeff(binx, biny, k) * hxton[k % 4] * hyton[k / 4];
409  sum += lsum;
410  }
411  }
412  // move from unit square coordinates to user coordinates
413  return sum * binSizeX * binSizeY;
414 }
