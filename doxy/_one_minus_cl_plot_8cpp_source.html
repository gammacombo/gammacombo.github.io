551 void OneMinusClPlot::drawSolutions()
552 {
553  m_mainCanvas->cd();
554  m_mainCanvas->Update();
555  float ymin = gPad->GetUymin();
556  float ymax = gPad->GetUymax();
557  float xmin = gPad->GetUxmin();
558  float xmax = gPad->GetUxmax();
559  int iDrawn = 0;
560 
561  for ( int i = 0; i < scanners.size(); i++ )
562  {
563  if ( scanners[i]->getDrawSolution()==0 ) continue;
564  if ( arg->debug ) cout << "OneMinusClPlot::drawSolutions() : adding solution for scanner " << i << " ..." << endl;
565  float xCentral = scanners[i]->getScanVar1Solution(arg->plotsoln[i]);
566  float xCLmin = scanners[i]->getCLinterval(arg->plotsoln[i]).min;
567  float xCLmax = scanners[i]->getCLinterval(arg->plotsoln[i]).max;
568  int color = scanners[i]->getTextColor();
569 
570  // draw vertical lines at central value and
571  // upper/lower errors
572  if ( ! arg->isQuickhack(19) ) {
573  drawVerticalLine(xCentral, color, kSolid);
574  if ( ! arg->isQuickhack(20) ) {
575  drawVerticalLine(xCLmin, color, kDashed);
576  drawVerticalLine(xCLmax, color, kDashed);
577  }
578  }
579 
580  // draw text box with numerical values after the lines,
581  // so that it doesn't get covered
582 
583  // compute y position of the printed central value
584  float yNumberMin = 0.6 - 0.13*(float)iDrawn;
585  float yNumberMax = yNumberMin+0.1;
586  if ( arg->plotlog ) {
587  float yNumberMinFirst = 0.1;
588  if ( arg->isQuickhack(1) ) yNumberMinFirst = 0.175;
589  yNumberMin = yNumberMinFirst / pow(3.0, iDrawn); // move down by a constant shift on log scale
590  yNumberMax = yNumberMin*2.;
591  }
592  // if printsoly option then move a bit
593  if ( arg->printSolY > 0. ) {
594  yNumberMin += arg->printSolY;
595  yNumberMax += arg->printSolY;
596  }
597 
598  // compute x position of the printed central value
599  float xNumberMin, xNumberMax;
600  if ( scanners[i]->getDrawSolution()==1 ) {
601  xNumberMin = xCentral+(xmax-xmin)*0.20; // draw at central value
602  xNumberMax = xCentral+(xmax-xmin)*0.0;
603  }
604  else if ( scanners[i]->getDrawSolution()==2 ) {
605  xNumberMin = xCLmin+(xmax-xmin)*0.0; // draw at left CL boundary
606  xNumberMax = xCLmin+(xmax-xmin)*0.25;
607  }
608  else if ( scanners[i]->getDrawSolution()==3 ) {
609  xNumberMin = xCLmax+(xmax-xmin)*0.0; // draw at right CL boundary
610  xNumberMax = xCLmax+(xmax-xmin)*0.25;
611  }
612  else if ( scanners[i]->getDrawSolution()==4 ) {
613  xNumberMin = xCLmin+(xmax-xmin)*-0.20; // draw a little left of the left CL boundary
614  xNumberMax = xCLmin+(xmax-xmin)*0.0;
615  }
616  else {
617  cout << "OneMinusClPlot::drawSolutions() : ERROR : --ps code ";
618  cout << scanners[i]->getDrawSolution() << " not found! Use [0,1,2,3]." << endl;
619  continue;
620  }
621 
622  // move number a bit to the left so it doesn't cover the right plot border anymore
623  if ( arg->isQuickhack(4) )
624  {
625  xNumberMin -= (xmax-xmin)*0.225;
626  xNumberMax -= (xmax-xmin)*0.225;
627  }
628 
629  // if print solution argument given then over write
630  if ( arg->printSolX > 0. )
631  {
632  float diff = xNumberMax - xNumberMin;
633  xNumberMin = arg->printSolX;
634  xNumberMax = arg->printSolX + diff;
635  }
636 
637  TPaveText *t1 = new TPaveText(xNumberMin, yNumberMin, xNumberMax, yNumberMax, "BR");
638  t1->SetBorderSize(0);
639  t1->SetFillStyle(0);
640  t1->SetTextAlign(13);
641  t1->SetTextFont(font);
642  t1->SetTextColor(color);
643  if ( isAngle(scanners[i]->getScanVar1()) ){
644  xCentral = RadToDeg(xCentral);
645  xCLmin = RadToDeg(xCLmin);
646  xCLmax = RadToDeg(xCLmax);
647  }
648  Rounder myRounder(arg, xCLmin, xCLmax, xCentral);
649  int d = myRounder.getNsubdigits();
650  float xCentralRd = myRounder.central();
651  if ( arg->isQuickhack(3) ) xCentralRd += 180.;
652  t1->AddText(Form("%.*f^{+%.*f}_{#font[122]{-}%.*f}",
653  d, xCentralRd,
654  d, myRounder.errPos(),
655  d, myRounder.errNeg()))->SetTextSize(labelsize);
656  t1->Draw();
657  iDrawn += 1;
658  }
