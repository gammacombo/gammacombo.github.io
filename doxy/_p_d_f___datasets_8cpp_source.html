374 void PDF_Datasets::initializeRandomGenerator(int seedShift) {
375 
376  if (seedShift == 0) {
377  // From the ROOT documentation:
378  // if seed is 0 [...] a TUUID is generated and used to fill the first 8 integers of the seed array.
379  // In this case the seed is guaranteed to be unique in space and time.
380  RooRandom::randomGenerator()->SetSeed(0);
381  } else {
382  // calculate unique seed for deterministic random generation
383  if (arg == NULL) {
384  std::cerr << "Error in PDF_Datasets::initializeRandomGenerator." << std::endl;
385  std::cerr << "You must pass the OptParser in the constructor in order to use this function." << std::endl;
386  exit(EXIT_FAILURE);
387  }
388  RooRandom::randomGenerator()->SetSeed(seedShift + (arg->nrun) * (arg->ntoys) * (arg->npoints1d));
389  }
390 }

392 void PDF_Datasets::unblind(TString var, TString unblindRegs) {
393 
394  TString unblindString = "";
395  TObjArray *regs = unblindRegs.Tokenize(","); // split string at ","
396  for (int i=0; i<regs->GetEntries(); i++){
397  TString range = ((TObjString*)regs->At(i))->GetString();
398  TString minStr = range;
399  TString maxStr = range;
400  minStr.ReplaceAll("[","");
401  minStr.Replace(minStr.Index(":"), minStr.Sizeof(), "");
402  maxStr.ReplaceAll("]","");
403  maxStr.Replace(0,maxStr.Index(":")+1,"");
404  float min = minStr.Atof();
405  float max = maxStr.Atof();
406  wspc->var(var)->setRange(Form("unblind%d",i),min,max);
407  unblindString += Form("unblind%d",i);
408  if (i < regs->GetEntries()-1) unblindString += ",";
409  }
410  unblindRegions[var] = unblindString;
