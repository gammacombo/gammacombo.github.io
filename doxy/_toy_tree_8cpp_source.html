393 void ToyTree::computeMinMaxN()
394 {
395  if ( scanpointN!=-1 ) return;
396  assert(t);
397  vector<float> pointsx;
398  vector<float> pointsy;
399  float _minx = 1e6;
400  float _maxx = -1e6;
401  float _miny = 1e6;
402  float _maxy = -1e6;
403  TObjArray* branches = t->GetListOfBranches();
404  t->SetBranchStatus("*", 0);
405  if(branches->FindObject("scanpoint")) t->SetBranchStatus("scanpoint", 1);
406  if(branches->FindObject("scanpointy")) t->SetBranchStatus("scanpointy", 1);
407  Long64_t nentries = t->GetEntries();
408  if ( nentries==0 ) return;
409  ProgressBar pb(arg, nentries);
410  if ( arg->debug ) cout << "ToyTree::computeMinMaxN() : ";
411  cout << "analysing toys ..." << endl;
412  for (Long64_t i = 0; i < nentries; i++){
413  // status bar
414  pb.progress();
415  t->GetEntry(i);
416  // Cut away toys outside a certain range. Also check line 1167 in MethodPluginScan.cpp.
417  if ( arg->pluginPlotRangeMin!=arg->pluginPlotRangeMax
418  && !(arg->pluginPlotRangeMin<scanpoint && scanpoint<arg->pluginPlotRangeMax) ) continue;
419  _minx = TMath::Min(_minx, scanpoint);
420  _maxx = TMath::Max(_maxx, scanpoint);
421  _miny = TMath::Min(_miny, scanpointy);
422  _maxy = TMath::Max(_maxy, scanpointy);
423  pointsx.push_back(scanpoint);
424  pointsy.push_back(scanpointy);
425  }
426  sort(pointsx.begin(), pointsx.end());
427  sort(pointsy.begin(), pointsy.end());
428  float binWidthx = -1;
429  float binWidthy = -1;
430  bool foundDifferentBinWidths = false;
431  float pointsPrevx = pointsx[0];
432  float pointsPrevy = pointsy[0];
433  int _nDifferentScanPointsx = 1;
434  int _nDifferentScanPointsy = 1;
435  for ( unsigned int i=1; i<pointsx.size(); i++ ) {
436  if ( pointsx[i] != pointsPrevx ) {
437  _nDifferentScanPointsx += 1;
438  if ( binWidthx == -1 ) binWidthx = fabs(pointsx[i]-pointsPrevx); // save first bin width so we can compare to others
439  }
440  if ( pointsy[i] != pointsPrevy ) {
441  _nDifferentScanPointsy += 1;
442  if ( binWidthy == -1 ) binWidthy = fabs(pointsy[i]-pointsPrevy); // save first bin width so we can compare to others
443  }
444  if ( binWidthx>-1 && fabs(pointsx[i]-pointsPrevx)>1e-6 && fabs(binWidthx-fabs(pointsx[i]-pointsPrevx))>1e-6 ) foundDifferentBinWidths = true;
445  if ( binWidthy>-1 && fabs(pointsy[i]-pointsPrevy)>1e-6 && fabs(binWidthy-fabs(pointsy[i]-pointsPrevy))>1e-6 ) foundDifferentBinWidths = true;
446  pointsPrevx = pointsx[i];
447  pointsPrevy = pointsy[i];
448  }
449  if ( arg->debug ) printf("ToyTree::computeMinMaxN() : min(x)=%f, max(x)=%f, n(x)=%i\n", _minx, _maxx, _nDifferentScanPointsx);
450  if ( arg->debug && arg->var.size()==2 ) printf("ToyTree::computeMinMaxN() : min(y)=%f, max(y)=%f, n(y)=%i\n", _miny, _maxy, _nDifferentScanPointsy);
451  if ( foundDifferentBinWidths ) {
452  cout << "\nToyTree::computeMinMaxN() : WARNING : Different bin widths found in the toys!" << endl;
453  cout << " The p-value histogram will have binning problems.\n" << endl;
454  }
455  scanpointMin = _minx;
456  scanpointMax = _maxx;
457  scanpointN = _nDifferentScanPointsx;
458  scanpointyMin = _miny;
459  scanpointyMax = _maxy;
460  scanpointyN = _nDifferentScanPointsy;
461  t->SetBranchStatus("*", 1);
462  open(); // this is a workaround to fix an issue where the branches get somehow disconnected by reconnecting them
463 }
