106 
107  // for the status bar
108  float nTotalSteps = nPoints1d;

120  {
121  int j = jj;
122  if ( reverse ) switch(jj)
123  {
124  case 0: j = 2; break;
125  case 1: j = 3; break;

138  scanUp = true;
139  break;
140  case 1:
141  // DOWN
142  scanStart = par->getMax();
143  scanStop = startValue;
144  scanUp = false;
145  break;
146  case 2:

183

186

209  warningChi2Neg = "MethodProbScan::scan1d() : WARNING : " + title;

211  warningChi2Neg += " setting to: " + TString(Form("%f",newChi2minScan));
212  //cout << warningChi2Neg << "\r" << flush;
213  cout << warningChi2Neg << endl;

218  // previous 1-CL values are too high.
219  if ( chi2minScan<chi2minGlobal ){

221  << " chi2minScan=" << chi2minScan << endl;

224  for ( int k=1; k<=hCL->GetNbinsX(); k++ ){

226  }
227  }
228

230  double oneMinusCL = TMath::Prob(deltaChi2, 1);
231 
232  // Save the 1-CL value and the corresponding fit result.
233  // But only if better than before!
234  if ( hCL->GetBinContent(hCL->FindBin(scanvalue)) <= oneMinusCL ){
235  hCL->SetBinContent(hCL->FindBin(scanvalue), oneMinusCL);
236  hChi2min->SetBinContent(hCL->FindBin(scanvalue), chi2minScan);
237  int iRes = hCL->FindBin(scanvalue)-1;
238  curveResults[iRes] = r;
239  }
240 
241  nStep++;
242  }
243  }
244  cout << "MethodProbScan::scan1d() : scan done. " << endl;
245 
246  if ( bestMinFoundInScan-bestMinOld > 0.01 ){
247  cout << "MethodProbScan::scan1d() : WARNING: Scan didn't find similar minimum to what was found before!" << endl;
248  cout << "MethodProbScan::scan1d() : Too strict parameter limits? Too coarse scan steps? Didn't load global minimum?" << endl;
249  cout << "MethodProbScan::scan1d() : chi2 bestMinFoundInScan=" << bestMinFoundInScan << ", bestMinOld=" << bestMinOld << endl;
250  }
251 
252  // attempt to correct for undercoverage
253  if (pvalueCorrectorSet) {
254  for ( int k=1; k<=hCL->GetNbinsX(); k++ ){
255  double pvalueProb = hCL->GetBinContent(k);
256  pvalueProb = pvalueCorrector->transform(pvalueProb);
257  hCL->SetBinContent(k, pvalueProb);
258  }
259  }
260 
261  setParameters(w, parsName, startPars->get(0));
262  saveSolutions();
263  if (arg->confirmsols) confirmSolutions();

266  return 0;

312 bool MethodProbScan::computeInnerTurnCoords(const int iStart, const int jStart,
313  const int i, const int j, int &iResult, int &jResult, int nTurn)
314 {
315  // compute bin coordinates of start parameters: connect center of
316  // the spiral to the scan point with a straight line, go back by sqrt(2)
317  // units, take bin this ends us in
318  iResult = iStart;
319  jResult = jStart;
320  if ( sq(i-iStart)+sq(j-jStart)>0 )
321  {
322  iResult = round((float)i - float(nTurn)*1.41 * float(i-iStart)/sqrt(sq(i-iStart)+sq(j-jStart)));
323  jResult = round((float)j - float(nTurn)*1.41 * float(j-jStart)/sqrt(sq(i-iStart)+sq(j-jStart)));
324  }
325  if ( iResult-1>=curveResults2d.size() ) iResult = iStart;
326  if ( jResult-1>=curveResults2d[0].size() ) jResult = jStart;
327  // check result
328  if ( iResult-1>=curveResults2d.size() || jResult-1>=curveResults2d[0].size()
329  || iResult-1<0 || jResult-1<0 )
330  {
331  cout << "MethodProbScan::computeInnerTurnCoords() : ERROR : resulting coordinates out of range! "
332  << iResult-1 << " " << jResult-1 << endl;
333  }
334  if ( iResult==iStart && jResult==jStart ) return false;
335  return true;
336 }

598 void MethodProbScan::saveSolutions()
599 {
600  if ( arg->debug ) cout << "MethodProbScan::saveSolutions() : searching for minima in hChi2min ..." << endl;
601 
602  // delete old solutions if any
603  vector<RooSlimFitResult*> tmp;
604  solutions = tmp;
605 
606  // loop over chi2 histogram to locate local maxima
607  for ( int i=2; i<hChi2min->GetNbinsX()-1; i++ )
608  {
609  bool oneBinMax = hChi2min->GetBinContent(i-1) > hChi2min->GetBinContent(i)
610  && hChi2min->GetBinContent(i+1) > hChi2min->GetBinContent(i);
611  bool twoBinMax = hChi2min->GetBinContent(i-1) > hChi2min->GetBinContent(i)
612  && hChi2min->GetBinContent(i )== hChi2min->GetBinContent(i+1)
613  && hChi2min->GetBinContent(i+2) > hChi2min->GetBinContent(i+1);
614  if ( !(oneBinMax || twoBinMax )) continue;
615 
616  // loop over fit results to find those that produced it
617  for ( int j=0; j<curveResults.size(); j++ ){
618  if ( !curveResults[j] ){
619  if ( arg->debug ) cout << "MethodProbScan::saveSolutions() : WARNING : empty solution at index " << j << endl;
620  continue;
621  }
622 
623  if ( hChi2min->FindBin(curveResults[j]->getConstParVal(scanVar1))==i ){
