240 void MethodDatasetsProbScan::sethCLFromProbScanTree() {
241  this->probScanTree->open();
242  float halfBinWidth = (this->probScanTree->getScanpointMax() - this->probScanTree->getScanpointMin()) / ((float)this->probScanTree->getScanpointN()) / 2; //-1.)/2;
248  delete hCL;
249  this->hCL = new TH1F("hCL", "hCL", this->probScanTree->getScanpointN(), this->probScanTree->getScanpointMin() - halfBinWidth, this->probScanTree->getScanpointMax() + halfBinWidth);
250  if (arg->debug) printf("DEBUG %i %f %f %f\n", this->probScanTree->getScanpointN(), this->probScanTree->getScanpointMin() - halfBinWidth, this->probScanTree->getScanpointMax() + halfBinWidth, halfBinWidth);
251  delete hCLs;
252  this->hCLs = new TH1F("hCLs", "hCLs", this->probScanTree->getScanpointN(), this->probScanTree->getScanpointMin() - halfBinWidth, this->probScanTree->getScanpointMax() + halfBinWidth);
253  if (arg->debug) printf("DEBUG %i %f %f %f\n", this->probScanTree->getScanpointN(), this->probScanTree->getScanpointMin() - halfBinWidth, this->probScanTree->getScanpointMax() + halfBinWidth, halfBinWidth);
254  Long64_t nentries = this->probScanTree->GetEntries();
255  this->probScanTree->activateAllBranches();
256  // this->probScanTree->a - DATASETSctivateCoreBranchesOnly(); //< speeds up the event loop
257  for (Long64_t i = 0; i < nentries; i++)
258  {
259  // load entry
260  this->probScanTree->GetEntry(i);
261 
262  double deltaChi2 = probScanTree->chi2min - probScanTree->chi2minGlobal;
263  double oneMinusCL = TMath::Prob( deltaChi2, 1);
264  // save the value and corresponding fit result
265  // but only if an improvement
266  if ( hCL->GetBinContent(hCL->FindBin( probScanTree->scanpoint ) ) <= oneMinusCL ) {
267  hCL->SetBinContent( hCL->FindBin( probScanTree->scanpoint ) , oneMinusCL );
268  hChi2min->SetBinContent( hCL->FindBin( probScanTree->scanpoint ), probScanTree->chi2min );
269  }
270  // and whilst we here do the same relative to the background only hypothesis (i.e. for CLs method)
271  double deltaChi2Bkg = probScanTree->chi2min - probScanTree->chi2minBkg;
272  //double oneMinusCLBkg = TMath::Prob( deltaChi2Bkg, 1);
273  double oneMinusCLBkg = getPValueTTestStatistic( deltaChi2Bkg, true );
274  if ( hCLs->GetBinCenter( hCLs->FindBin( probScanTree->scanpoint ) ) <= oneMinusCLBkg ) {
275  hCLs->SetBinContent( hCLs->FindBin( probScanTree->scanpoint ), oneMinusCLBkg );
276  }
277  }
278  // if only the plot method then set the global and bkg minimums from the file
279  if ( arg->isAction("plot") ) {
280  chi2minGlobal = probScanTree->chi2minGlobal;
281  chi2minBkg = probScanTree->chi2minBkg;
282  }
283 
284  // put in best fit value
285  hCL->SetBinContent(hCL->FindBin( probScanTree->scanbest ),1.);
286  hChi2min->SetBinContent(hCL->FindBin( probScanTree->scanbest),chi2minGlobal);
287  hCLs->SetBinContent(hCLs->FindBin( probScanTree->scanbest ), 1.);
288 
289  cout << "Best fit at: scanVar = " << probScanTree->scanbest << " with Chi2Min: " << chi2minGlobal << endl;
290 
291  sortSolutions();
292  //saveSolutions();
293  // this->probScanTree->activateAllBranches(); //< Very important!
294  //

361 int MethodDatasetsProbScan::scan1d(bool fast, bool reverse)
362 {
363  if (fast) return 0; // tmp
364 
365  if ( arg->debug ) cout << "MethodDatasetsProbScan::scan1d() : starting ... " << endl;
366 
367  // Set limit to all parameters.
368  this->loadParameterLimits();
369 
370 
371  // Define scan parameter and scan range.
372  RooRealVar *parameterToScan = w->var(scanVar1);
373  float parameterToScan_min = hCL->GetXaxis()->GetXmin();
374  float parameterToScan_max = hCL->GetXaxis()->GetXmax();
375 
376  // do a free fit
377  RooFitResult *result = this->loadAndFit(this->pdf); // fit on data
378  assert(result);
379  RooSlimFitResult *slimresult = new RooSlimFitResult(result,true);
380  slimresult->setConfirmed(true);
381  solutions.push_back(slimresult);
382  double freeDataFitValue = w->var(scanVar1)->getVal();
383 
384  // Define outputfile
385  system("mkdir -p root");
386  TString probResName = Form("root/scan1dDatasetsProb_" + this->pdf->getName() + "_%ip" + "_" + scanVar1 + ".root", arg->npoints1d);
387  TFile* outputFile = new TFile(probResName, "RECREATE");
388 
389  // Set up toy root tree
390  this->probScanTree = new ToyTree(this->pdf, arg);
391  this->probScanTree->init();
392  this->probScanTree->nrun = -999; //\todo: why does this branch even exist in the output tree of the prob scan?
393 
394  // Save parameter values that were active at function
395  // call. We'll reset them at the end to be transparent
396  // to the outside.
397  RooDataSet* parsFunctionCall = new RooDataSet("parsFunctionCall", "parsFunctionCall", *w->set(pdf->getParName()));
398  parsFunctionCall->add(*w->set(pdf->getParName()));
399 
400  // start scan
401  cout << "MethodDatasetsProbScan::scan1d_prob() : starting ... with " << nPoints1d << " scanpoints..." << endl;
402  ProgressBar progressBar(arg, nPoints1d);
403  for ( int i = 0; i < nPoints1d; i++ )
404  {
405  progressBar.progress();
406  // scanpoint is calculated using min, max, which are the hCL x-Axis limits set in this->initScan()
407  // this uses the "scan" range, as expected
408  // don't add half the bin size. try to solve this within plotting method
409 
410  float scanpoint = parameterToScan_min + (parameterToScan_max - parameterToScan_min) * (double)i / ((double)nPoints1d - 1);
411  if (arg->debug) cout << "DEBUG in MethodDatasetsProbScan::scan1d_prob() " << scanpoint << " " << parameterToScan_min << " " << parameterToScan_max << endl;
412 
413  this->probScanTree->scanpoint = scanpoint;
414 
415  if (arg->debug) cout << "DEBUG in MethodDatasetsProbScan::scan1d_prob() - scanpoint in step " << i << " : " << scanpoint << endl;
416 
417  // don't scan in unphysical region
418  // by default this means checking against "free" range
419  if ( scanpoint < parameterToScan->getMin() || scanpoint > parameterToScan->getMax() + 2e-13 ) {
420  cout << "it seems we are scanning in an unphysical region: " << scanpoint << " < " << parameterToScan->getMin() << " or " << scanpoint << " > " << parameterToScan->getMax() + 2e-13 << endl;
421  exit(EXIT_FAILURE);
422  }
423 
424  // FIT TO REAL DATA WITH FIXED HYPOTHESIS(=SCANPOINT).
425  // THIS GIVES THE NUMERATOR FOR THE PROFILE LIKELIHOOD AT THE GIVEN HYPOTHESIS
426  // THE RESULTING NUISANCE PARAMETERS TOGETHER WITH THE GIVEN HYPOTHESIS ARE ALSO
427  // USED WHEN SIMULATING THE TOY DATA FOR THE FELDMAN-COUSINS METHOD FOR THIS HYPOTHESIS(=SCANPOINT)
428  // Here the scanvar has to be fixed -> this is done once per scanpoint
429  // and provides the scanner with the DeltaChi2 for the data as reference
430  // additionally the nuisances are set to the resulting fit values
431 
432  parameterToScan->setVal(scanpoint);
433  parameterToScan->setConstant(true);
434 
435  RooFitResult *result = this->loadAndFit(this->pdf); // fit on data
436  assert(result);
437 
438  if (arg->debug) {
439  cout << "DEBUG in MethodDatasetsProbScan::scan1d_prob() - minNll data scan at scan point " << scanpoint << " : " << 2 * result->minNll() << ": "<< 2 * pdf->getMinNll() << endl;
440  }
441  this->probScanTree->statusScanData = result->status();
442 
443  // set chi2 of fixed fit: scan fit on data
444  // CAVEAT: chi2min from fitresult gives incompatible results to chi2min from pdf
445  // this->probScanTree->chi2min = 2 * result->minNll();
446  this->probScanTree->chi2min = 2 * pdf->getMinNll();
447  this->probScanTree->covQualScanData = result->covQual();
448  this->probScanTree->scanbest = freeDataFitValue;
449 
450  // After doing the fit with the parameter of interest constrained to the scanpoint,
451  // we are now saving the fit values of the nuisance parameters. These values will be
452  // used to generate toys according to the PLUGIN method.
453  this->probScanTree->storeParsScan(); // \todo : figure out which one of these is semantically the right one
454 
455  this->pdf->deleteNLL();
456 
457  // also save the chi2 of the free data fit to the tree:
458  this->probScanTree->chi2minGlobal = this->getChi2minGlobal();
459  this->probScanTree->chi2minBkg = this->getChi2minBkg();
460 
461  this->probScanTree->genericProbPValue = this->getPValueTTestStatistic(this->probScanTree->chi2min - this->probScanTree->chi2minGlobal);
462  this->probScanTree->fill();
463 
464  if(arg->debug && pdf->getBkgPdf())
465  {
466  float pval_cls = this->getPValueTTestStatistic(this->probScanTree->chi2min - this->probScanTree->chi2minBkg, true);
467  cout << "DEBUG in MethodDatasetsProbScan::scan1d() - p value CLs: " << pval_cls << endl;
468  }
469 
470 
471  // reset
472  setParameters(w, pdf->getParName(), parsFunctionCall->get(0));
473  //setParameters(w, pdf->getObsName(), obsDataset->get(0));
474  } // End of npoints loop
475  probScanTree->writeToFile();
476  if (bkgOnlyFitResult) bkgOnlyFitResult->Write();
477  if (dataFreeFitResult) dataFreeFitResult->Write();
478  outputFile->Close();
479  std::cout << "Wrote ToyTree to file" << std::endl;
480  delete parsFunctionCall;
481 
482  // This is kind of a hack. The effect is supposed to be the same as callincg
483  // this->sethCLFromProbScanTree(); here, but the latter gives a segfault somehow....
484  // \todo: use this->sethCLFromProbScanTree() directly after figuring out the cause of the segfault.
485  this->loadScanFromFile();
486 
487  return 0;
488 }
