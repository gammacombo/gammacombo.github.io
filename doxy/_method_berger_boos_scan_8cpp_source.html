50 TH2F* MethodBergerBoosScan::calcPValues(TH2F better, TH2F all, TH2F bg){
51  TH2F* cl = (TH2F*) better.Clone("cl");
52  for (int i=1; i<=better.GetNbinsX(); i++)
53  {
54  for (int j = 1; j <= better.GetNbinsY(); ++j)
55  {
56  float nbetter = better.GetBinContent(i,j);
57  float nall = all.GetBinContent(i,j);
58  float nbackground = bg.GetBinContent(i,j);
59  if ( nall == 0. ) continue;
60 
61  // subtract background
62  // float p = (nbetter-nbackground)/(nall-nbackground);
63  // hCL->SetBinContent(i, p);
64  // hCL->SetBinError(i, sqrt(p * (1.-p)/(nall-nbackground)));
65 
66  // don't subtract background
67  float p = nbetter/nall;
68  cl->SetBinContent(i,j, p);
69  cl->SetBinError(i,j, sqrt(p * (1.-p)/nall));
70  }
71  }
72  return cl;
