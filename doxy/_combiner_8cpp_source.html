137 void Combiner::combine()
138 {
139  if ( _isCombined ){
140  cout << "Combiner::combine() : WARNING : Already combined. Skipping." << endl;
141  return;
142  }
143  if ( arg->debug ) cout << "Combiner::combine() : combining name=" << name << " title=" << title << endl;
144  if ( pdfs.size()==0 ){
145  cout << "Combiner::combine() : ERROR : Combination is empty." << endl;
146  return;
147  }
148  // check for pitfalls and warn the user
149  for (int i=0; i<pdfs.size(); i++ ){
150  for (int j=i+1; j<pdfs.size(); j++ ){
151  PDF_Abs* p = pdfs[j];
152  if ( pdfs[i]->getUniqueGlobalID() == p->getUniqueGlobalID() ){
153  cout << "\nWARNING : You are trying to combine the same PDF twice!" << endl;
154  cout << " Ignore this warning if you're doing it intentionally!" << endl;
155  cout << " combiner: " << title << endl;
156  cout << " PDF: " << p->getBaseName() << endl;
157  }
158  else if ( pdfs[i]->getBaseName() == p->getBaseName() ){
159  cout << "\nWARNING : You are trying to combine two PDFs with the same name." << endl;
160  cout << " Ignore this warning if you're doing it intentionally!" << endl;
161  cout << " combiner: " << title << endl;
162  cout << " PDF: " << p->getBaseName() << endl;
163  }
164  }
165  }
166 
167  // uniquify all input pdfs, add them to the workspace
168  for (int i=0; i<pdfs.size(); i++ ){
169  if ( arg->debug ) cout << "Combiner::combine() : processing PDF " << pdfs[i]->getName() << endl;
170  // check consistency of input pdfs
171  bool pdfOk = pdfs[i]->checkConsistency();
172  if ( !pdfOk ){
173  cout << "Combiner::combine() : ERROR : PDF " << pdfs[i]->getName() << " is not self consitent. Exit." << endl;
174  exit(1);
175  }
176  // uniquify pdf
177  pdfs[i]->uniquify(i); // Needs to be unique inside this combiner, not globally: it's important
178  // that same combiner's pdfs are named the same
179  // else we can't save toys from different combiners into
180  // the same ToyTree in the coverage test.
181  // Also, the "scan for observable" mechanism relies on the fact
182  // that the ID coincides with the number of the PDF in this combiner.
183  // add PDF to workspace
184  RooMsgService::instance().setGlobalKillBelow(WARNING);
185  if ( pdfs[i]->isCrossCorPdf() ){
186  // cross correlation PDFs need the same observable names as the main PDFs,
187  // so link them together in the workspace
188  w->import(*pdfs[i]->getPdf(),RecycleConflictNodes());
189  }
190  else{
191  w->import(*pdfs[i]->getPdf());
192  }
193  w->defineSet("obs_"+pdfs[i]->getName(), *pdfs[i]->getObservables());
194  w->defineSet("par_"+pdfs[i]->getName(), *pdfs[i]->getParameters());
195  w->defineSet("th_" +pdfs[i]->getName(), *pdfs[i]->getTheory());
196  RooMsgService::instance().setGlobalKillBelow(INFO);
197  // save unique pdf name
198  pdfNames.push_back((pdfs[i]->getName()).Data());
199  }
200 
201  // sort pdfs alphabetically
202  sort( pdfNames.begin(), pdfNames.end() );
203 
204  // combine
205  pdfName = pdfNames[0];
206  for (int i=1; i<pdfNames.size(); i++ )
207  {
208  // build name of combined pdf
209  TString newName = pdfName+"_"+pdfNames[i];
210  if ( w->pdf("pdf_"+newName) ) continue;
211 
212  // combine pdfs by multiplying them one by one:
213  w->factory("PROD::pdf_"+newName+"(pdf_"+pdfName+", pdf_"+pdfNames[i]+")");
214 
215  // define sets of combined parameters
216  parsName = "par_"+newName;
217  obsName = "obs_"+newName;
218  mergeNamedSets(w, parsName, "par_"+pdfName, "par_"+pdfNames[i]);
219  mergeNamedSets(w, obsName, "obs_"+pdfName, "obs_"+pdfNames[i]);
220  mergeNamedSets(w, "th_"+newName, "th_" +pdfName, "th_" +pdfNames[i]);
221  pdfName = newName;
222  }
223  setParametersConstant();
224  _isCombined = true;
225 }
