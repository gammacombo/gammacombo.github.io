104  double c = fitParams[3];
105  y = ( x + 0.5*a*x*x + (b/c)*(1.-exp(-1.*c*x)) ) / ( 1. + 0.5*a + (b/c)*(1.-exp(-1.*c)) );
106  }
107  if (transFunc=="p1+1/x") {
108  double a = fitParams[1]/fitParams[0];
109  double b = fitParams[2]/fitParams[0];
110  double c = fitParams[3];
