601 double MethodCoverageScan::transform( vector<double> fitParams, TString transFunc, double x )
602 {
603  double y = -999.; // this should be getting returned
604 
605  if ( transFunc!="p1" && transFunc!="p1+exp" && transFunc!="p1+1/x" ) {
606  cout << "ERROR -- " << transFunc << " is not a valid function" << endl;
607  exit(1);
608  }
609  if (transFunc=="p1") assert(fitParams.size()==2);
610  if (transFunc=="p1+exp") assert(fitParams.size()==4);
611  if (transFunc=="p1+1/x") assert(fitParams.size()==4);
612 
613 
614  if (transFunc=="p1") {
615  double a = fitParams[1]/fitParams[0];
616  y = ( x + 0.5*a*x*x ) / ( 1. + 0.5*a );
617  }
618  if (transFunc=="p1+exp") {
619  double a = fitParams[1]/fitParams[0];
620  double b = fitParams[2]/fitParams[0];
621  double c = fitParams[3];
622  y = ( x + 0.5*a*x*x + (b/c)*(1.-exp(-1.*c*x)) ) / ( 1. + 0.5*a + (b/c)*(1.-exp(-1.*c)) );
623  }
624  if (transFunc=="p1+1/x") {
625  double a = fitParams[1]/fitParams[0];
626  double b = fitParams[2]/fitParams[0];
627  double c = fitParams[3];
628  y = ( x + 0.5*a*x*x + b*log((x+c)/c) ) / ( 1. + 0.5*a + b*log((x+c)/c) );
629  }
630 
631  assert( y >= 0. && y <= 1.);
632  return y;
633 }
